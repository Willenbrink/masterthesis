\section{Inference Rules}
\inference[T-TASK]{x:C; \mathit{ocap} \vdash t : \tau \\ \Gamma; a \vdash b : Q \vartriangleright \mathit{Box}[C]}{\Gamma; a \vdash \mathit{task}(b) \{x \Rightarrow t\} : Q \vartriangleright \mathit{Task}[C]}
\inference[]{}{}

\inference[T-ASYNC]{\mathit{Perm}[Q] \in \Gamma \\ \Gamma \setminus \mathit{Perm}[Q]; a \vdash s : \sigma \\ \Gamma; a \vdash t : Q \vartriangleright \mathit{Task}[C]}{\Gamma; a \vdash \mathit{async}(t) \{s\} : \bot}
\inference[]{}{}

\inference[T-FINISH]{\Gamma; a \vdash t : \tau}{\Gamma; a \vdash \mathit{finish} \{ t \} : null}
\inference[]{}{}

\NewDocumentCommand{\sframe}{ s O{L} m O{P} O{l} }{\langle #2, \IfBooleanT{#1}{&&} #3,\ \IfBooleanT{#1}{&&} #4 \IfBooleanT{#1}{&&}\rangle^#5 \IfBooleanT{#1}{&&} }
\NewDocumentCommand{\fframe}{ s m }{\IfBooleanT{#1}{&&} \langle \mathit{FINISH} #2 \rangle}
\NewDocumentCommand{\state}{ O{H} O{f} m O{TS} }{#1, \{(#2, #3)\} && \uplus #4}
\NewDocumentCommand{\reducframe}{m m}{%
  \begin{alignedat}{7}
    &#1\\\rightarrow\ &#2
  \end{alignedat}%
}
\NewDocumentCommand{\reducstack}{m m}{%
  \begin{alignedat}{7}
    &#1\\\twoheadrightarrow\ &#2
  \end{alignedat}%
}



\inference[E-TASK]{L(b') = b(o, p)}{%
  \reducframe{%
      H, \sframe*{%
        let\ x = \mathit{task}(b')\{x \Rightarrow t\} in\ s%
      }%
  }{%
      H, \sframe*[L[x \rightarrow \mathit{task}(b(o,p),t)]]{%
        s
      }%
  }%
}
\inference[]{}{}

\inference[E-ASYNC]{%
  \nexists \fframe{\_} \in FS\\
  L(y) = \mathit{task}(b(o,p),t) \\
  T = (f', \sframe[[x \rightarrow o]]{t}[\emptyset][\epsilon])\\
}{%
  \reducstack{%
    \state{%
      \sframe*{\mathit{async}(y)\{s\}}[p \uplus P] \circ FS \circ \fframe*{f'} \circ FS'
    }}{%
    \state{%
      \sframe*{s}[P] \circ \fframe*{f'} \circ FS'%
    }[\{T\} \uplus TS]%
  }%
}
\inference[]{}{}

\inference[E-FINISH1]{%
  F_1 = \sframe{t}[P][\epsilon]\\
  F_2 = \fframe{f'}\\
  F_3 = \sframe{null}\\
  f' \mathit{fresh}\\
}{%
  \reducstack{%
    \state{%
      \sframe{%
        \mathit{finish} \{\ t\ \}%
      } \circ FS%
    }
  }{%
    \state{%
      F_1 \circ F_2 \circ F_3 \circ FS%
    }
  }
}
\inference[]{}{}

\inference[E-FINISH2]{%
  \nexists (f', FS) \in TS
}{%
  \reducstack{%
    \state{%
      \fframe{f'} \circ FS
    }
  }{%
    \state{%
      FS
    }
  }
}
